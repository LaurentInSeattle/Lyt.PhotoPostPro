Fancy Sharpening 

Duplicate the Layer: Make a copy of your image layer so you can work non-destructively.

Detect Edges: Select the duplicated layer and run a Find Edges (or Detect Edges) filter. This will isolate 
the outlines of your subjects

Convert to Black and White: Desaturate this layer and increase the contrast (using a Levels adjustment) 
until the edges are distinct white lines against a solid black background.

Soften the Mask: Apply a slight Gaussian Blur to the mask. This prevents the final sharpening from looking too harsh 
or "crunchy".

Apply Sharpening: Add a standard Unsharp Mask to your original photo. 

Load the black-and-white edge mask you created as an alpha channel/mask to apply the effect only to the edges.

	1. The Photographic Exposure EquationExposure is the mathematical product of three camera variables
	(the Exposure Triangle):\(\text{Exposure} \propto \text{Aperture Area} \times \text{Shutter Speed} \times \text{ISO Sensitivity}\)
	To change the exposure by a specific number of stops, you adjust any of these three values:
	Aperture (f-stop): Controls the size of the lens opening. The area is proportional to 
	\(\frac{1}{N^{2}}\) (where \(N\) is the f-number). 
	To let in twice as much light (or add \(+1 \text{ stop}\)), you multiply the aperture denominator 
	by \(\frac{1}{\sqrt{2}}\) (approx \(0.7\)). 
	Common full stops include:\(f/1.4 \rightarrow f/2 \rightarrow f/2.8 \rightarrow f/4 \rightarrow f/5.6 \rightarrow f/8 \rightarrow f/11 \rightarrow f/16 \rightarrow f/22\)
	Shutter Speed (\(t\)): Controls the duration of light exposure. 
	Measured in fractions of a second, doubling the time (e.g., from \(\frac{1}{60}\) to \(\frac{1}{30}\)) adds \(+1 \text{ stop}\) of light.ISO (\(S\)): Controls sensor sensitivity. Doubling the ISO (e.g., \(100 \rightarrow 200\)) adds \(+1 \text{ stop}\) of exposure.2. Math for Post-Processing (Digital Editing)If you are adjusting a photo's exposure in editing software like Photoshop or Lightroom, the math scales linearly with pixel values.For an 8-bit image (where pixel values range from \(0\) to \(255\)), you apply the following formula:\(\text{New Pixel Value} = \text{Original Pixel Value} \times 2^{\Delta EV}\)Where \(\Delta EV\) is the change in Exposure Value (stops).Example: To increase the exposure by \(+1 \text{ stop}\) (\(\Delta EV = 1\)), multiply the original pixel brightness by \(2^{1}\) (\(2\)).Example: To decrease exposure by \(-1 \text{ stop}\) (\(\Delta EV = -1\)), multiply by \(2^{-1}\) (\(0.5\)).3. Exposure Value (EV) FormulaIf you want to calculate the exact mathematical Exposure Value (EV) of a scene, 
	you use a logarithmic equation relating your settings to scene brightness:\(EV = \log_2 \left( \frac{N^2}{t} \right)\)
	
	Where:\(N\) is the aperture f-number\(t\) is the shutter speed in seconds

public class LutGenerator

	// Generates a 32x32x32 Identity LUT
	public static Color[,,] CreateIdentityLut(int size = 32)
	{
		Color[,,] lut = new Color[size, size, size];
		for (int b = 0; b < size; b++)
		{
			for (int g = 0; g < size; g++)
			{
				for (int r = 0; r < size; r++)
				{
					int rValue = (r * 255) / (size - 1);
					int gValue = (g * 255) / (size - 1);
					int bValue = (b * 255) / (size - 1);
					lut[r, g, b] = Color.FromArgb(rValue, gValue, bValue);
				}
			}
		}
		return lut;
	}

					labPixels[idx, 0] = lab.L;
					labPixels[idx, 1] = lab.A;

					sumL += lab.L;
					sumA += lab.A;
					sumB += lab.B;
					idx++;
				}
			}
		});

		ImageStats stats = new ImageStats
		{
			MeanL = sumL / count,
			MeanA = sumA / count,
			MeanB = sumB / count
		};

		double varL = 0, varA = 0, varB = 0;
		for (int i = 0; i < count; i++)
		{
			varL += Math.Pow(labPixels[i, 0] - stats.MeanL, 2);
			varA += Math.Pow(labPixels[i, 1] - stats.MeanA, 2);
			varB += Math.Pow(labPixels[i, 2] - stats.MeanB, 2);
		}

		// Guard against division by zero on flat/solid color images
		stats.StdL = Math.Max(0.0001, Math.Sqrt(varL / count));
		stats.StdA = Math.Max(0.0001, Math.Sqrt(varA / count));
		stats.StdB = Math.Max(0.0001, Math.Sqrt(varB / count));

		return stats;
	}

	private static Rgb24 MatchColor(Rgb24 srcColor, ImageStats src, ImageStats tgt)
	{
		// Convert input grid point color to Lab
		var (l, a, b) = RgbToLab(srcColor);

		// Scale and shift based on standard deviation ratios and means
		l = ((l - src.MeanL) * (tgt.StdL / src.StdL)) + tgt.MeanL;
		a = ((a - src.MeanA) * (tgt.StdA / src.StdA)) + tgt.MeanA;
		b = ((b - src.MeanB) * (tgt.StdB / src.StdB)) + tgt.MeanB;

		return LabToRgb(l, a, b);
	}

	// Color conversion pipeline: RGB -> XYZ -> LMS -> log10(LMS) -> lab
	private static (double L, double A, double B) RgbToLab(Rgb24 rgb)
	{
		double r = rgb.R / 255.0;
		double g = rgb.G / 255.0;
		double b = rgb.B / 255.0;

		// RGB to LMS matrix multiplication
		double L = (0.3811 * r) + (0.5783 * g) + (0.0402 * b);
		double M = (0.1967 * r) + (0.7244 * g) + (0.0789 * b);
		double S = (0.0241 * r) + (0.1288 * g) + (0.8444 * b);

		// Log space transform (handle zeros gracefully)
		L = L > 0 ? Math.Log10(L) : -5.0;
		M = M > 0 ? Math.Log10(M) : -5.0;
		S = S > 0 ? Math.Log10(S) : -5.0;

		// LMS to Ruderman lab space matrix multiplication
		double labL = (L + M + S) / Math.Sqrt(3);
		double labA = (L + M - (2 * S)) / Math.Sqrt(6);
		double labB = (L - (2 * M)) / Math.Sqrt(2);

		return (labL, labA, labB);
	}

	// Color conversion pipeline: lab -> log10(LMS) -> LMS -> XYZ -> RGB
	private static Rgb24 LabToRgb(double labL, double labA, double labB)
	{
		// Inverse Ruderman matrix multiplication
		double L = (labL / Math.Sqrt(3)) + (labA / Math.Sqrt(6)) + (labB / Math.Sqrt(2));
		double M = (labL / Math.Sqrt(3)) + (labA / Math.Sqrt(6)) - (labB / Math.Sqrt(2));
		double S = (labL / Math.Sqrt(3)) - (2 * labA / Math.Sqrt(6));

		// Inverse Log transform
		L = Math.Pow(10, L);
		M = Math.Pow(10, M);
		S = Math.Pow(10, S);

		// LMS to RGB matrix multiplication
		double r = (4.4679 * L) - (3.5873 * M) + (0.1193 * S);
		double g = (-1.2186 * L) + (2.3809 * M) - (0.1624 * S);
		double b = (0.0497 * L) - (0.2439 * M) + (1.2045 * S);

		// Clamp values safely into valid byte boundaries
		byte resR = (byte)Math.Clamp(r * 255.0, 0, 255);
		byte resG = (byte)Math.Clamp(g * 255.0, 0, 255);
		byte resB = (byte)Math.Clamp(b * 255.0, 0, 255);

		return new Rgb24(resR, resG, resB);
	}

	private static Rgb24[,,] CreateIdentityLut(int size)
	{
		Rgb24[,,] lut = new Rgb24[size, size, size];
		for (int b = 0; b < size; b++)

			for (int g = 0; g < size; g++)

				for (int r = 0; r < size; r++)
				{
					byte rVal = (byte)((r * 255) / (size - 1));
					byte gVal = (byte)((g * 255) / (size - 1));
					byte bVal = (byte)((b * 255) / (size - 1));
					lut[r, g, b] = new Rgb24(rVal, gVal, bVal);
				}
			}
		}
		return lut;
	}
